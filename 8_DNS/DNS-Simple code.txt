import java.io.*;
import java.util.*;
import java.net.*;

class DNS1 {
   public static void main(String[] args)throws Exception {
      try {
         InetAddress[] addr = InetAddress.getAllByName("google.co.i");
         for(int i = 0; i < addr.length; i++)
            System.out.println(addr[i]);
      }
      
      catch(Exception e) {
         System.out.println("Cannot access information: "+e);
      }
   }
}
